          <p style="margin-top: 0;"><strong>บริษัท สปาร์ค แฟคเทอร์ จำกัด ("บริษัท")</strong> ให้ความสำคัญอย่างยิ่งยวดกับการคุ้มครองข้อมูลส่วนบุคคลของท่าน บริษัทจึงจัดทำเอกสารฉบับนี้ขึ้นเพื่อขอความยินยอมในการเก็บรวบรวม ใช้ และเปิดเผยข้อมูลส่วนบุคคลของท่าน ตามพระราชบัญญัติคุ้มครองข้อมูลส่วนบุคคล พ.ศ. 2562 (PDPA) โดยมีรายละเอียดดังต่อไปนี้</p>

          

          <p><strong>1. ข้อมูลส่วนบุคคลที่บริษัทเก็บรวบรวม</strong><br>

          บริษัทมีความจำเป็นต้องเก็บรวบรวมข้อมูลส่วนบุคคลของท่านที่ได้ให้ไว้ผ่านแบบสอบถามนี้ ได้แก่</p>

          <ul style="margin-top: 5px; margin-bottom: 15px; padding-left: 20px;">

            <li>ข้อมูลระบุตัวตนและข้อมูลการติดต่อ เช่น ชื่อ-นามสกุล, หมายเลขโทรศัพท์ และที่อยู่อีเมล</li>

            <li>ข้อมูลเกี่ยวกับโครงสร้างธุรกิจ ปัญหา และเป้าหมายทางการตลาดของท่าน</li>

          </ul>


          <p><strong>2. วัตถุประสงค์ในการเก็บรวบรวมและการใช้ข้อมูล</strong><br>

          บริษัทจะนำข้อมูลส่วนบุคคลของท่านไปใช้เพื่อวัตถุประสงค์ดังต่อไปนี้เท่านั้น:</p>

          <ul style="margin-top: 5px; margin-bottom: 15px; padding-left: 20px;">

            <li>เพื่อประมวลผลและจัดส่งผลลัพธ์การวิเคราะห์แบบประเมินแนวทางการตลาดให้แก่ท่านผ่านทางอีเมล</li>

            <li>เพื่อให้เจ้าหน้าที่ของบริษัทติดต่อกลับเพื่อนำเสนอข้อมูล ให้คำปรึกษา หรือเสนอผลิตภัณฑ์และบริการที่สอดคล้องกับความต้องการของท่าน</li>

            <li>เพื่อนำข้อมูลไปวิเคราะห์เชิงสถิติและพัฒนาปรับปรุงบริการของบริษัทให้มีประสิทธิภาพมากยิ่งขึ้น</li>

          </ul>


          <p><strong>3. การเปิดเผยข้อมูลส่วนบุคคล</strong><br>

          บริษัทจะเก็บรักษาข้อมูลของท่านไว้เป็นความลับ และจะไม่มีการเปิดเผยข้อมูลส่วนบุคคลของท่านให้แก่บุคคลภายนอก เว้นแต่จะได้รับความยินยอมจากท่าน หรือเป็นการปฏิบัติตามที่กฎหมายกำหนด</p>


          <p><strong>4. ระยะเวลาในการเก็บรักษาข้อมูล</strong><br>

          บริษัทจะเก็บรักษาข้อมูลส่วนบุคคลของท่านไว้เป็นระยะเวลา 2 ปี นับตั้งแต่วันที่ท่านให้ข้อมูล หรือจนกว่าท่านจะมีการร้องขอให้ลบทำลายข้อมูล หรือเพิกถอนความยินยอมตามสิทธิของท่าน</p>


          <p><strong>5. สิทธิของเจ้าของข้อมูลส่วนบุคคล</strong><br>

          <ul style="margin-top: 5px; margin-bottom: 0; padding-left: 20px;">

            <li>สิทธิในการขอเข้าถึงและขอรับสำเนาข้อมูลส่วนบุคคลของท่าน</li>

            <li>สิทธิในการขอแก้ไขข้อมูลให้ถูกต้อง เป็นปัจจุบัน และสมบูรณ์</li>

            <li>สิทธิในการขอให้ลบ ทำลาย หรือระงับการใช้ข้อมูลส่วนบุคคล</li>

            <li>สิทธิในการเพิกถอนความยินยอมที่ได้ให้ไว้เมื่อใดก็ได้ โดยติดต่อมายังบริษัทผ่านช่องทางติดต่อที่บริษัทกำหนด</li>

      // ฟังก์ชันสำหรับปลดล็อกฟอร์มเมื่อกดยอมรับ PDPA

      function toggleForm() {

        const isChecked = document.getElementById('pdpaConsent').checked;

        const mainForm = document.getElementById('mainFormContent');

        

        if (isChecked) {

          mainForm.disabled = false; // ปลดล็อกฟอร์มให้กรอกได้

        } else {

          mainForm.disabled = true;  // ล็อกฟอร์มถ้าเอาติ๊กออก

        }

      }


      // ฟังก์ชันสำหรับเปิด/ปิด ช่องระบุธุรกิจ "อื่นๆ"

      function toggleOtherBusiness() {

        const businessType = document.getElementById('businessType').value;

        const otherSection = document.getElementById('otherBusinessSection');

        const otherInput = document.getElementById('otherBusiness');

        

        if (businessType === "H. อื่น ๆ") {

          otherSection.style.display = "block"; // แสดงช่องกรอก

          otherInput.required = true;           // บังคับว่าต้องพิมพ์ข้อความ

        } else {

          otherSection.style.display = "none";  // ซ่อนช่องกรอก

          otherInput.required = false;          // ปลดบังคับกรอก

          otherInput.value = "";                // เคลียร์ค่าที่พิมพ์ค้างไว้

        }

      }


      // ฟังก์ชันสำหรับประมวลผลเงื่อนไขการแสดงช่องกรอกข้อมูลติดต่อ

      function toggleContactFields() {

        const emailRadio = document.querySelector('input[name="wantEmail"]:checked');

        const contactRadio = document.querySelector('input[name="wantContact"]:checked');

        

        // ดึงค่าว่าลูกค้าเลือกอะไร (ถ้ายังไม่เลือกจะให้เป็นค่าว่าง)

        const valEmail = emailRadio ? emailRadio.value : "";

        const valContact = contactRadio ? contactRadio.value : "";


        const nameEmailSection = document.getElementById('nameEmailSection');

        const phoneSection = document.getElementById('phoneSection');

        

        const nameInput = document.getElementById('name');

        const emailInput = document.getElementById('email');

        const phoneInput = document.getElementById('phone');


        // เงื่อนไขที่ 1: การแสดงกล่อง "ชื่อ และ อีเมล" 

        // (จะแสดงก็ต่อเมื่อ Want Email เป็น "ต้องการ" หรือ Want Contact เป็น "ต้องการ")

        if (valEmail === "ต้องการ" || valContact === "ต้องการ") {

          nameEmailSection.style.display = "block";

          nameInput.required = true;

          emailInput.required = true;

        } else {

          nameEmailSection.style.display = "none";

          nameInput.required = false;

          emailInput.required = false;

          nameInput.value = "";   // เคลียร์ข้อมูลทิ้ง

          emailInput.value = "";

        }


        // เงื่อนไขที่ 2: การแสดงกล่อง "เบอร์โทร"

        // (จะแสดงก็ต่อเมื่อ Want Contact เป็น "ต้องการ" เท่านั้น)

        if (valContact === "ต้องการ") {

          phoneSection.style.display = "block";

          phoneInput.required = true;

        } else {

          phoneSection.style.display = "none";

          phoneInput.required = false;

          phoneInput.value = "";  // เคลียร์ข้อมูลทิ้ง

        }

      }
